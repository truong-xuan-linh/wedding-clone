SMTP_HOST=smtp.gmail.com                                                                                        
SMTP_PORT=587                                                                                                   
SMTP_USER=email-cua-ban@gmail.com                                                                               
SMTP_PASS=app-password-gmail         # bật 2FA rồi tạo App Password                                             
OWNER_EMAIL=email-nhan-thong-bao@gmail.com
